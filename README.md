const int listSizes[] = {97,94,98,98,96,97,96,74,94,88};

#include <Wire.h>

//initialize the liquid crystal library
//the first parameter is  the I2C address
//the second parameter is how many rows are on your screen
//the  third parameter is how many columns are on your screen
LiquidCrystal_I2C lcd(0x27,  16, 2);

static volatile int state = 1;

#define BootUp 1
#define Generate 2
#define Distribute 3
#define Wait 4
#define End 5

int players = 3;
int currentPlayer = 0;
int numImposters = 1;
bool imposters[100];

volatile bool secondInterrupt;

bool categorySelected[] = {true,true,true,true,true,true,true,true,true,true};

int randEntryIdx;
int randListIdx;

void pickRandomEntry() {
  int selectedIndices[10];
  int selectedCount = 0;

  for (int i = 0; i < 10; i++) {
    if (categorySelected[i]) {
      selectedIndices[selectedCount++] = i;
    }
  }

  if (selectedCount == 0) {
    lcd.clear();
    lcd.setCursor(0,0);
    lcd.print(F("Waehle mindestens"));
    lcd.setCursor(0,1);
    lcd.print(F("eine Kategorie"));
    while(digitalRead(11));
    menuRender();
    state = BootUp;
  }

  // Select random list and entry
  randListIdx = selectedIndices[random(selectedCount)];
  int listSize = listSizes[randListIdx];
  randEntryIdx = random(listSize);
}




void setup() {
  randomSeed(analogRead(A0));
  //initialize lcd screen
  lcd.init();
  // turn on the backlight
  lcd.backlight();
  pinMode(12,INPUT_PULLUP);
  pinMode(13,INPUT_PULLUP);
  pinMode(3,INPUT_PULLUP);
  lcd.setCursor(0,0);
  lcd.clear();
  Serial.begin(115200);
  menuRender();
  set_sleep_mode(SLEEP_MODE_PWR_DOWN);
}
void loop() {
  //Serial.println(test);
  //Serial.println(digitalRead(12));
  switch(state){
    case BootUp: bootUp(); break;
    case Generate: generate(); break;
    case Distribute: distribute(); break;
    case Wait: wait(); break; 
    case End: end(); break;
  }
}

bool lastLeft;
bool lastMiddle;
bool lastRight;

#define numLines 12

int currentLine = 0;
bool menuDepth = false;
//String category[] = {"around_the_world", "entertainment", "dailyLife", "animals_and_nature", "sports_and_leisure", "knowledge_and_school", "festivals_and_celebrations", "stars_and_celebrities", "german_terms", "spicy" };
const String category[] = {"Um die Welt", "Unterhaltung", "Alltag", "Tiere + Natur", "Sport", "Schule", "Feste", "Stars", "Begriffe", "Spicy" };

void bootUp(){
  bool l = !digitalRead(13);
  bool m = !digitalRead(12);
  bool r = !digitalRead(3);
  bool l_pressed = l && ! lastLeft;
  bool m_pressed = m && ! lastMiddle;
  bool r_pressed = r && ! lastRight;
  lastLeft = l;
  lastMiddle = m;
  lastRight = r;
  if(m_pressed){
    if(menuDepth){
      menuDepth = false;
    }else{
      changeDepth(currentLine);
    }
  }
  if(menuDepth){
    if(l_pressed){
      if(currentLine == 10){
        players++;
      }else if(currentLine==11){
        numImposters++;
      }
    }
    if(r_pressed){
      if(currentLine == 10){
        players--;
      }else if(currentLine==11){
        numImposters--;
      }
    }
  }else{
    if(l_pressed){
      currentLine--;
      if(currentLine<0)
        currentLine = numLines;
    }
    if(r_pressed){
      currentLine++;
      currentLine = currentLine % (numLines+1);
    }
  }
  if(l_pressed||r_pressed||m_pressed){
    menuRender();
  }
  delay(10);
}

void menuRender(){
  lcd.clear();
  lcd.setCursor(0,0);
  lcd.print(getLine(currentLine));
  lcd.setCursor(0,1);
  lcd.print(getLine(currentLine+1));
}

void changeDepth(byte line){
  if(currentLine < 10){
    categorySelected[line] = !categorySelected[line];
  }
  switch(line){
    case 10: menuDepth = true;break;
    case 11: menuDepth = true;break;
    case 12: state = Generate;break;
  }
}

String getLine(byte line){
  String text = getLineText(line);
  String value = getLineValue(line);
  String result = text;
  for(int i = 0;i < 16-(text.length()+value.length());i++){
     result += " ";
  }
  return result + value;
}
String getLineText(byte line){
  if(line < 10){
    return category[line];
  }
  switch(line){
    case 10: return "Spieler";
    case 11: return "Imposter";
    case 12: return "Start";
    case 13: return "";
  }
}
String getLineValue(byte line){
    if(line < 10){
    return (categorySelected[line] ? "J" : "N");
  }
  switch(line){
    case 10: return String(players);
    case 11: return String(numImposters);
    case 12: return "";
    case 13: return "";
  }
}

void generate(){
  for (int x = 0; x < players; x++)
  {
    imposters[x] = false;
  }
  for (int x = 0; x < numImposters; x++)
  {
    int imposter;
    do{
      imposter = random(0,players);
    }while(imposters[imposter]);
    imposters[imposter] = true;
  } 
  currentPlayer = 0;
  pickRandomEntry();
  diplay();
  secondInterrupt = false;
  state = Distribute;
  delay(100);
}
void distribute(){
  if(digitalRead(3)){
    lcd.noDisplay();
  }else{
    lcd.display();
  }
  if(!digitalRead(12)){
    currentPlayer++;
    if(currentPlayer >= players){
      state = Wait;
      return;
    }
    lcd.noBacklight();
    while(!digitalRead(12)){
     delay(10); 
    }
    lcd.backlight();
    diplay();
    delay(100);
  }
}
boolean testImposter(int test)
{
  return imposters[test];
}
void diplay(){
  lcd.setCursor(0,0);
  lcd.clear();
  boolean im = testImposter(currentPlayer);
  lcd.print(im?"Imposter":"Player");
  lcd.setCursor(0,1);
  char buffer[23];  
  
  // Get pointer to list in PROGMEM
  const char* const* listPtr = (im) ? (const char* const*)pgm_read_ptr(&(hint_lists[randListIdx])) : (const char* const*)pgm_read_ptr(&(word_lists[randListIdx]));
  const char* entryPtr = (const char*)pgm_read_ptr(&(listPtr[randEntryIdx]));

  // Copy entry from PROGMEM to buffer
  strcpy_P(buffer, entryPtr);

  lcd.print(String(buffer));
}
void wait(){
  while(!digitalRead(3));
  delay(10);
  lcd.clear();
  lcd.noBacklight();
  sleep_enable();
  attachInterrupt(digitalPinToInterrupt(3), isr, FALLING);
  delay(10);
  sleep_cpu();
}
volatile void isr(){
  if(secondInterrupt){
    state = End;
  }
  else
  {
    secondInterrupt = true;
  }
}
void end(){sleep_disable();
  detachInterrupt(digitalPinToInterrupt(3));
  lcd.backlight();
  lcd.clear();
  lcd.display();
  lcd.setCursor(0,0);
  lcd.print("Imposter: ");
  lcd.setCursor(0,1);
  for (int x = 0; x < players; x++)
  {
    if(imposters[x]){
      Serial.println(x);
      lcd.print(String(x+1));
      lcd.print(" ");
    }
  } 
  while(digitalRead(12));
  state = Generate;
  while(!digitalRead(12));
}
